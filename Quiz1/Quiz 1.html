# Write your answer here:
data = pd.read_csv('data_quiz/wdi_ageprop_2020.csv')
list_varnames = ["percent_ages0to14","percent_ages15to64"]

fig, list_subfig = plt.subplots(1, 2,figsize = (6,3))
plt.tight_layout()
index = 0
for var in list_varnames:
    list_subfig[index].hist(data[var])
    list_subfig[index].set_title("Distribution for " + var)
    list_subfig[index].set_ylabel("Frequency")
    list_subfig[index].set_xlabel(var)
    index = index + 1
